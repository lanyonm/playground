001package org.lanyonm.playground.web.controller;

004import static org.springframework.restdocs.mockmvc.MockMvcRestDocumentation.document;

008import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.model;
009import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.redirectedUrl;
010import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.status;
011import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.view;
012
013import java.util.List;
014
015import org.junit.Before;
016import org.junit.Rule;
017import org.junit.Test;
018import org.junit.runner.RunWith;
019import org.lanyonm.playground.config.DataConfig;
020import org.lanyonm.playground.config.ViewResolver;
021import org.lanyonm.playground.config.WebConfig;
022import org.lanyonm.playground.domain.Todo;
023import org.lanyonm.playground.service.ExceptionServiceImpl;
024import org.lanyonm.playground.service.TodoServiceImpl;
025import org.lanyonm.playground.service.UserServiceImpl;
026import org.springframework.beans.factory.annotation.Autowired;
027import org.springframework.restdocs.RestDocumentation;
028import org.springframework.test.context.ContextConfiguration;
029import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
030import org.springframework.test.context.web.AnnotationConfigWebContextLoader;
031import org.springframework.test.context.web.WebAppConfiguration;
032import org.springframework.test.web.servlet.MockMvc;
033import org.springframework.test.web.servlet.MvcResult;
034import org.springframework.test.web.servlet.setup.MockMvcBuilders;
035import org.springframework.web.context.WebApplicationContext;
036
037/**
038 * @author lanyonm
039 */
040@RunWith(SpringJUnit4ClassRunner.class)
041@WebAppConfiguration
042@ContextConfiguration(loader=AnnotationConfigWebContextLoader.class, classes={DataConfig.class, ExceptionServiceImpl.class, TodoServiceImpl.class, UserServiceImpl.class, ViewResolver.class, WebConfig.class})
043public class TodoControllerTest {
044
045        @Rule
046        public final RestDocumentation restDocumentation = new RestDocumentation("target/generated-snippets");
047
048        @Autowired
049        private WebApplicationContext wac;
050
051        private MockMvc mockMvc;
052
053        @Before
054        public void setup() {
055                this.mockMvc = MockMvcBuilders.webAppContextSetup(this.wac)
056                                .apply(documentationConfiguration(this.restDocumentation))
057                                .build();
058        }
059
060        @Test
061        public void testIndex() throws Exception {
062                this.mockMvc.perform(get("/todo/")).andExpect(status().isOk())
063                                .andExpect(view().name("todo/index"))
064                                .andExpect(model().attributeExists("todoList"))
065                                .andDo(document("todo"));
066        }
067
068        @Test
069        public void testTodoAddEditShow() throws Exception {
070                this.mockMvc.perform(get("/todo/{id}/edit", Integer.MAX_VALUE))
071                                .andExpect(status().is(302))
072                                .andExpect(redirectedUrl("/todo/"));
073                this.mockMvc.perform(get("/todo/{id}/edit", 0))
074                                .andExpect(status().isOk())
075                                .andExpect(view().name("todo/addEdit"))
076                                .andExpect(model().attribute("todo", new Todo()));
077                this.mockMvc.perform(get("/todo/{id}/edit", 1))
078                                .andExpect(status().isOk())
079                                .andExpect(view().name("todo/addEdit"))
080                                .andExpect(model().attributeExists("todo"));
081        }
082
083        @Test
084        public void testTodoEdit() throws Exception {
085                // TODO add coverage for nefarious edit
086                this.mockMvc.perform(post("/todo/{id}/edit", 1).param("title", "furst tudo"))
087                                .andExpect(status().is(302))
088                                .andExpect(redirectedUrl("/todo/"));
089        }
090
091        @SuppressWarnings("unchecked")
092        @Test
093        public void testTodoAddDelete() throws Exception {
094                this.mockMvc.perform(post("/todo/{id}/edit", 0).param("title", "second todo"))
095                                .andExpect(redirectedUrl("/todo/"));
096                MvcResult result = this.mockMvc.perform(get("/todo/"))
097                                .andExpect(status().isOk())
098                                .andExpect(model().attributeExists("todoList"))
099                                .andReturn();
100                List<Todo> todoList = (List<Todo>) result.getModelAndView().getModel().get("todoList");
101                assertEquals("there should be two todos", todoList.size(), 2);
102                long todoId = todoList.get(todoList.size() - 1).getId();
103                this.mockMvc.perform(get("/todo/{id}/delete", todoId))
104                                .andExpect(status().is(302))
105                                .andExpect(redirectedUrl("/todo/"));
106                result = this.mockMvc.perform(get("/todo/"))
107                                .andExpect(status().isOk())
108                                .andExpect(model().attributeExists("todoList"))
109                                .andReturn();
110                todoList = (List<Todo>) result.getModelAndView().getModel().get("todoList");
111                assertEquals("there should be one todo", todoList.size(), 1);
112        }
113}
